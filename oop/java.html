Word Extraction

String s="New York University has one of the ugliest college campuses in the country";
String w[];
int i,nw,k=0;
s=s.trim()+" ";//to make sure that we don't get an IndexOutOfBoundsException
for(i = 0 ; i < s.length() ;) //for finding the string's length
{
i+=s.substring(i,s.indexOf(" ",i).length()+1;

        nw++;

}
w=new String[nw]; for(i = 0 ; i < s.length() ;) //now we're gonna extract the words
{
        w[k]=s.substring(i,s.indexOf(" ",i);

        k++

        i+=w.length()+1;

        nw++;

}
